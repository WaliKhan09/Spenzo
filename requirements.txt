# INSTALLATION NOTES

# 
# To install all dependencies:
# npm install
# 
# To install specific categories:
# npm install @clerk/nextjs @arcjet/next  # Authentication & Security
# npm install @prisma/client prisma       # Database & ORM
# npm install @google/generative-ai       # AI Services
# npm install @radix-ui/react-*           # UI Components
# npm install recharts lucide-react       # Data Visualization & Icons
# 
# =============================================================================
# ENVIRONMENT VARIABLES REQUIRED
# =============================================================================
# 
# Database:
# DATABASE_URL=postgresql://username:password@localhost:5432/spenzo
# DIRECT_URL=postgresql://username:password@localhost:5432/spenzo
# 
# Authentication (Clerk):
# NEXT_PUBLIC_CLERK_PUBLISHABLE_KEY=your_clerk_publishable_key
# CLERK_SECRET_KEY=your_clerk_secret_key
# 
# AI Services:
# GOOGLE_AI_API_KEY=your_google_ai_api_key
# 
# Email (Resend):
# RESEND_API_KEY=your_resend_api_key
# 
# Background Jobs (Inngest):
# INNGEST_EVENT_KEY=your_inngest_event_key
# INNGEST_SIGNING_KEY=your_inngest_signing_key
# 
# =============================================================================
# SETUP COMMANDS
# =============================================================================
# 
# 1. Install dependencies:
#    npm install
# 
# 2. Generate Prisma client:
#    npx prisma generate
# 
# 3. Push database schema:
#    npx prisma db push
# 
# 4. Run development server:
#    npm run dev
